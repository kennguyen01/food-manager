4a. C switch statement

```c
#include <stdio.h>

int main() {
  int num = 2;
  
  switch(num) {
    case 1:
      printf("Choice is 1");
      break;
    case 2:
      printf("Choice is 2");
      break;
    default:
      printf("Choice other than 1 and 2");
      break;
  }
  return 0;
}
```
